// CREATE FINAL PDF // ============================= const finalPdf = await PDFLib.PDFDocument.create(); for (let i = 1; i <= reportPdf.numPages; i++) { const rendered = await renderPdfPageToImage(reportPdf, i, 2); const page = finalPdf.addPage([ rendered.width, rendered.height ]); // Report image const reportImage = await finalPdf.embedPng(rendered.image); page.drawImage(reportImage, { x: 0, y: 0, width: rendered.width, height: rendered.height }); // Letterhead overlay const lhImage = await finalPdf.embedPng(lhRendered.image); page.drawImage(lhImage, { x: 0, y: 0, width: rendered.width, height: rendered.height }); } const finalBytes = await finalPdf.save(); const blob = new Blob([ finalBytes ], { type: 'application/pdf' }); const url = URL.createObjectURL(blob); const link = document.createElement('a'); link.href = url; link.download = 'Final_Report_On_Letterhead.pdf'; document.body.appendChild(link); link.click(); document.body.removeChild(link); setTimeout(() => { URL.revokeObjectURL(url); }, 1000); } catch (error) { console.error(error); showError( 'Failed to process PDF.\n\nTry opening the report and using Print → Save as PDF, then upload again.' ); } finally { btn.disabled = false; btn.innerText = 'Generate Final PDF'; setLoading(false); } }
